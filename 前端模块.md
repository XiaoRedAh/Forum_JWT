```js
//制定校验规则

    { type: 'email', message: '请输入合法的电子邮箱地址', trigger: ['blur', 'change'] }
  ],
  code: [
    { required: true, message: '请输入获取的验证码', trigger: 'blur' },
    {min: 6, max: 6, message: '请输入6位验证码', trigger: ['blur', 'change']}
  ]
}

const coldTime = ref(0) //发送验证码的冷却时间
//判断邮箱地址是否有效（默认无效），有效才能激发“获取验证码”按钮
const isEmailValid = ref(false)
const onValidate= (prop, isValid) =>{
  //如果更新的属性是email
  if(prop === 'email')
    isEmailValid.value = isValid//isValid返回的是该属性是否校验通过
}

const emailFormRef = ref()

/*"修改电子邮箱"表单的各个参数*/
const emailForm = reactive({
  email: '',
  code: ''
})

//发送验证码
const sendCode = ()=>{
  emailFormRef.value.validate((isValid)=>{
    coldTime.value = 60
    if(isValid){
      get(`/api/auth/ask-code?email=${emailForm.email}&type=modify`, ()=>{
        ElMessage.success(`验证码已发送到邮箱: ${emailForm.email}，请注意查收`)
        const handle = setInterval(()=> {
          coldTime.value--
          if(coldTime.value === 0)
            clearInterval(handle)
        }, 1000)
      },(message) =>{
        ElMessage.warning(message)
        coldTime.value = 0 //验证码发送失败，冷却时间直接归0
      })
    }else{
      ElMessage.warning('表单内容有误，请重新检查表单内容')
    }
  })
}

//发送修改电子邮箱请求
const modifyEmail = ()=>{
  emailFormRef.value.validate((isValid)=>{
    if(isValid){
      post('/api/user/modify-email', {
        email: emailForm.email,
        code: emailForm.code
      }, (message)=>{
        ElMessage.success('修改电子邮箱成功')
        store.user.email = emailForm.email //store也要同步更新
        emailForm.code = ''
      })
    }else{
      ElMessage.warning('表单内容有误，请重新检查表单内容')
    }
  })
}
</script>
```

```js
  email: false,
  phone: false,
  qq: false,
  wx: false,

//展示现在有的隐私

}
```

## 重置密码

src/pages/settings/PrivacySetting.vue

js部分
```js
<script setup>
const formRef = ref()
const valid = ref(false)
const onValidate = (prop, isValid) => valid.value = isValid
const changePawForm = reactive({
  password_old: '',
  password_new: '',
  password_new_repeat: ''
})

const changePassword = ()=>{
  formRef.value.validate((valid)=>{
    if(valid){
      post('/api/user/change-password',{
        password_old: changePawForm.password_old,
        password_new: changePawForm.password_new
      },()=>{
        ElMessage.success("密码修改成功，请重新登录")
        //直接使用分装好的logout函数退出登录
        logout(()=>router.push('/'))
      })
    }else{
      ElMessage.warning("密码校验失败，请检查是否正确填写")
    }
  })
}

const validatePassword = (rule, value, callback) => {
  if (value === '') {
    callback(new Error('请再次输入密码'));
  } else if (value !== changePawForm.password_new) {
    callback(new Error('两次输入密码不一致!'));
  } else {
    callback();
  }
};

//制定校验规则
const rules ={
  password_old: [
    { required: true, message: '请输入密码', trigger: 'blur' },
    { min: 6, max: 16, message: '密码长度必须在6~16个字符之间', trigger: ['blur'] }
  ],
  password_new: [
    { required: true, message: '请输入密码', trigger: 'blur' },
    { min: 6, max: 16, message: '密码长度必须在6~16个字符之间', trigger: ['blur'] }
  ],
  password_new_repeat: [
    { required: true, message: '请再次输入密码', trigger: 'blur' },
    { validator: validatePassword, trigger: ['blur','change'] },
  ],
}
</script>
```

```js
function beforeAvatarUpload(rawFile){
  if(rawFile.type !== 'image/jpeg' && rawFile.type !== 'image/png'){

  }else if(rawFile.size / 1024 >1024){

}
```

# 帖子广场

帖子广场页面的布局构思：
* 分为左右两边内容，左侧占主要
* 左侧从上到下依次有编辑帖子的提示框，置顶帖子，帖子列表
  * 点击编辑帖子提示框，弹出“新增主题”卡片
* 右侧展示一些卡片，从上到下依次是论坛公告，天气信息，日期+ip，友情链接

## 新增主题卡片搭建

>src/components/TopicEditor.vue

这里使用到的富文本编辑器是**quill**，官网地址：https://vueup.github.io/vue-quill/
使用前需要先把依赖安装
```bash
npm install @vueup/vue-quill
```

使用el-drawer弹窗的细节
* `direction="btt"`：定义弹出的方向，使得弹窗自下而上地弹出
* `:close-on-click-modal="false"`：点击弹窗外的内容时，不会关闭弹窗，防止用户误操作

在使用第三方封装好的el-drawer，quill-editor等组件时，有些样式不合想法的，通过**deep样式穿刺**，强制改为自己想要的样子

如果文章字数不限制，可能会有很多超级大的文章存入系统，对服务器的压力是比较大的，因此在编辑文章时，需要**动态地统计字数**，限制字数在一定范围以内以内。

**跨组件控制**弹窗的弹出与关闭：ref变量 + defineprops + defineemit
